distmap(G=(V,E), s ∈∈ V):
– ∀∀ v ∈∈ V, dist[v] = ∞∞
– dist[s] = 0
– todo = [s]
– while todo ≠≠ []:
------ x = todo.pop_front()
------ for y ∈∈ δ+δ+(x):
---------- if (dist[y] == ∞∞):
------------ dist[y] <- dist[x] + 1
------------ todo.push_back(y)
– return dist

sortie: dist[y] la distance entre s et y
